    <script>
        document.addEventListener('DOMContentLoaded', async () => {
            const imageContainer = document.getElementById('image-container');
            
            try {
                // The prompt for the image generation, emphasizing data flow *between* a robot head and a human head
                const prompt = "A hyper-realistic, futuristic close-up image depicting a stylized **glowing robot head (representing AI) on the left side, facing right, with a subtle, illuminated mouth area**, and a **warm, contemplative human head on the right side, facing left, with a subtle, illuminated mouth area**. The focus is on the space directly between these two heads. **Numerous, bright, ultra-fine, pulsating light dots and glowing data packets rapidly flow and connect specifically within this central area, forming a concentrated, dynamic, visible pathway of knowledge and information transfer.** These lights should visually originate from the robot head and flow directly towards the human head, creating a powerful sense of focused, seamless learning and electrical impulses. The background should be subtly dark and abstract, indicating an advanced, intelligent learning environment, with soft light glows emanating from the heads themselves to highlight the data flow. The aesthetic is clean, highly detailed, and deeply luminous, with a strong emphasis on the seamless, one-way interconnectedness of knowledge flowing from AI to human within a defined central channel. The dominant colors should be deep electric blues, vibrant glowing greens, and ethereal, bright whites for the data pulses, conveying clarity and intellectual exchange, all concentrated in the central flow area.";
                
                // Construct the payload for the image generation API
                const payload = {
                    instances: { prompt: prompt },
                    parameters: { "sampleCount": 1 }
                };

                // Define the API URL for image generation
                const apiKey = ""; // API key will be provided by the Canvas runtime
                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/imagen-3.0-generate-002:predict?key=${apiKey}`;

                // Make the fetch call to the image generation API
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                // Parse the JSON response
                const result = await await response.json();

                // Check if image data is available in the response
                if (result.predictions && result.predictions.length > 0 && result.predictions[0].bytesBase64Encoded) {
                    const imageUrl = `data:image/png;base64,${result.predictions[0].bytesBase64Encoded}`;
                    // Add the image and multiple data pulse overlays for a dynamic effect
                    imageContainer.innerHTML = `
                        <img src="${imageUrl}" alt="Mesterséges intelligencia az oktatásban" class="rounded-lg shadow-lg">
                        <div class="data-pulse data-pulse-one"></div>
                        <div class="data-pulse data-pulse-two"></div>
                        <div class="data-pulse data-pulse-three"></div>
                        <div class="data-pulse data-pulse-four"></div>
                        <div class="data-pulse data-pulse-five"></div>
                        <div class="data-pulse data-pulse-six"></div>
                        <div class="data-pulse data-pulse-seven"></div>
                        <div class="data-pulse data-pulse-eight"></div>
                        <div class="data-pulse data-pulse-nine"></div>
                        <div class="data-pulse data-pulse-ten"></div>
                        <div class="data-pulse data-pulse-eleven"></div>
                        <div class="data-pulse data-pulse-twelve"></div>
                        <div class="data-pulse data-pulse-thirteen"></div>
                        <div class="data-pulse data-pulse-fourteen"></div>
                        <div class="data-pulse data-pulse-fifteen"></div>
                        <div class="mouth-glow robot-mouth-glow"></div>
                        <div class="mouth-glow human-mouth-glow"></div>
                    `;
                } else {
                    // Display an error message if image generation fails
                    imageContainer.innerHTML = `<p class="text-red-500">Nem sikerült képet generálni. Kérem, próbálja újra.</p>`;
                    console.error("Image generation failed:", result);
                }
            } catch (error) {
                // Catch and display any network or other errors
                imageContainer.innerHTML = `<p class="text-red-500">Hiba történt a kép betöltésekor: ${error.message}</p>`;
                console.error("Error fetching image:", error);
            }
        });
    </script>
